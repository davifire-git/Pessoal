# Regra geral: permita tudo, bloqueie só o que não deve aparecer no Google.


# ===== TODOS OS ROBÔS (regra padrão) =====
User-agent: *

# ===== GOOGLEBOT (indexação de páginas) =====

# ===== GOOGLE MOBILE BOT =====
# Avalia como o site aparece em dispositivos móveis

Allow: /

# ===== SLURP (Yahoo Search) =====
